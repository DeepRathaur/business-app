import { EventEmitter } from "events";
import { LocalStorageEnum } from "../enum/local-storage.enum";
import { localStorageService } from "./local-storage.service";
import {
  UMSPermissions,
  UserTypeEnum,
} from "@/shared/enum/ums-user-status.enum";
import { AccountDetailsModel, LayoutResponse, TransactionInfoModel, userPermission } from "@/shared/models";
class AccountService {
  private static logoutEmitter = new EventEmitter();
  private static loginStatusEmitter = new EventEmitter();
  static onLogout(callback: () => void) {
    this.logoutEmitter.on("logout", callback);
  }

  static onLoginStatusChange(callback: (status: boolean) => void) {
    this.loginStatusEmitter.on("loginStatus", callback);
  }

  isUserLoggedIn(): boolean {
    const token = localStorageService.get(LocalStorageEnum.ACCESS_LITE_TOK_EN);
    // if (!token) this.logoutEmitter.emit("logout");
    return !!token;
  }

  setAccessToken(token: string) {
    localStorageService.set(LocalStorageEnum.ACCESS_LITE_TOK_EN, token);
    // this.loginStatusEmitter.emit("loginStatus", true);
  }

  getAccessToken(): string | null {
    return localStorageService.get(LocalStorageEnum.ACCESS_LITE_TOK_EN);
  }

  setAccountNo(accountNo: string) {
    localStorageService.set(LocalStorageEnum.ACCOUNT_NO, accountNo);
  }

  getAccountNo(): string | null {
    return localStorageService.get(LocalStorageEnum.ACCOUNT_NO);
  }

  setPermission(permission: userPermission[]) {
    localStorageService.set(
      LocalStorageEnum.PERMISSION,
      JSON.stringify(permission)
    );
  }

  getPermission(): userPermission[] {
    const data = localStorageService.get(LocalStorageEnum.PERMISSION);
    return typeof data === "string" ? JSON.parse(data) : [];
  }

  setUser(user: any) {
    localStorageService.set(LocalStorageEnum.USER, JSON.stringify(user));
  }

  getUser(): any {
    const data = localStorageService.get(LocalStorageEnum.USER);
    return typeof data === "string" ? JSON.parse(data) : null;
  }

  logOut() {
    localStorage.clear();
    sessionStorage.clear();
  }

  setLoginTime(time: number) {
    localStorage.setItem(LocalStorageEnum.LOGIN_TIME, time.toString());
  }

  getLoginTime(): number | null {
    const time = localStorage.getItem(LocalStorageEnum.LOGIN_TIME);
    return time ? parseInt(time) : null;
  }

  setAuuid(auuid: string) {
    localStorageService.set<string>(LocalStorageEnum.AUU_NUMBERID, auuid);
  }

  getAuuid(): string | null {
    return localStorageService.get<string>(LocalStorageEnum.AUU_NUMBERID);
  }

  setServiceType(permission: string[]) {
    localStorageService.set<string[]>(
      LocalStorageEnum.SERVICE_TYPE,
      permission
    );
  }

  getServiceType() {
    return localStorageService.get<Array<string>>(
      LocalStorageEnum.SERVICE_TYPE
    );
  }

  setEnterPriseRole(roles: string) {
    localStorageService.set<string>(LocalStorageEnum.ENTERPRISE_ROLE, roles);
  }

  getEnterPriseRole(): UserTypeEnum | null {
    return localStorageService.get<UserTypeEnum>(
      LocalStorageEnum.ENTERPRISE_ROLE
    );
  }

  setConfigurationLayout(data: LayoutResponse) {
    localStorageService.set<string>(
      LocalStorageEnum.LAYOUT_CONF,
      JSON.stringify(data)
    );
  }

  getConfigurationLayout() {
    return localStorageService.get<LayoutResponse>(
      LocalStorageEnum.LAYOUT_CONF
    );
  }

  setUMSPermission(
    umsPermissions: UMSPermissions[],
    authorityNames: string[] = []
  ) {
    const permissionsToStore =
      authorityNames.length > 0 ? authorityNames : umsPermissions;
    localStorageService.set<string[]>(
      LocalStorageEnum.UMS_PERMISSION,
      permissionsToStore
    );
  }

  getUMSPermission() {
    return localStorageService.get<Array<UMSPermissions>>(
      LocalStorageEnum.UMS_PERMISSION
    );
  }

  setTxnInfo(data: TransactionInfoModel) {
    localStorageService.set<string>(LocalStorageEnum.TXN_INFO, JSON.stringify(data));

    const data = localStorageService.get(LocalStorageEnum.TXN_INFO);

  }

  getCurrentAccountDetails(): AccountDetailsModel {
    return localStorageService.get<AccountDetailsModel>(LocalStorageEnum.ACCOUNT_DETAIL) as AccountDetailsModel;
  }
}

import { LocalStorageEnum } from '../enum/local-storage.enum';
import { CryptoService } from './CipherService';

class LocalStorageService {
  private _crypto = new CryptoService();
  private _isEncrypted = process.env.NEXT_PUBLIC_ENCRYPTION === 'true';

  set<T>(key: LocalStorageEnum, value: T): void {
    if (typeof window === 'undefined') return;

    try {
      const storedValue = this._isEncrypted
        ? this._crypto.encryptData(value, key) ?? ''
        : JSON.stringify(value);

      localStorage.setItem(key, storedValue);
    } catch (error) {
      console.error(`Error setting localStorage key "${key}":`, error);
    }
  }

  get<T>(key: LocalStorageEnum): T | null {
    if (typeof window === 'undefined') return null;

    try {
      const raw = localStorage.getItem(key);
      if (!raw) return null;

      return this._isEncrypted
        ? this._crypto.decryptData(raw, key)
        : JSON.parse(raw);
    } catch (error) {
      console.error(`Error getting localStorage key "${key}":`, error);
      return null;
    }
  }

  clearAll(): void {
    if (typeof window === 'undefined') return;
    localStorage.clear();
  }

  clearItem(key: LocalStorageEnum): void {
    if (typeof window === 'undefined') return;
    localStorage.removeItem(key);
  }
}

export const localStorageService = new LocalStorageService();






import * as CryptoJS from 'crypto-js';
import forge from 'node-forge';
import { PemLoaderService } from './PamLoaderService';

export class CryptoService {

    private encryptSecretKey: string;
    private pemFileUrl: string = '/config/public-key.pem'; // Place in public 
    private _pemLoaderService = new PemLoaderService()


    constructor() {
        this.encryptSecretKey = typeof window !== 'undefined' ? `${window.navigator.appVersion}${window.navigator.appCodeName}${window.navigator.platform}${window.navigator.vendor}${window.navigator.userAgent}` : '';
    }

    private _readPemFile(file: File): Promise<string> {
        return new Promise((resolve, reject) => {
        const reader = new FileReader();
    
        reader.onload = event => {
            resolve(event.target?.result as string);
        };
        reader.onerror = error => {
            reject(error);
        };
        reader.readAsText(file);
        });
    }

    async encrypt(data: string): Promise<string> {
        return new Promise((resolve,reject)=>{

        this._pemLoaderService.loadPemFile(this.pemFileUrl).then(async (content: BlobPart) => {
            try {
            // Process the PEM content here
            const file = new File([content], 'mypublickey.pem');
        
            const pemFileContent = await this._readPemFile(file);
                // Convert PEM string to a forge public key
            const publicKey = forge.pki.publicKeyFromPem(pemFileContent as forge.pki.PEM);
            // Encrypt the data using RSA  
            const encrypted = publicKey.encrypt(data);

            // Convert the encrypted data to a base64 string  
            resolve(forge.util.encode64(encrypted));
            } catch (e) {
            console.log(e);
            reject();
            }
            
        },
        (error: any) => {
            console.error('Error loading PEM file:', error);
            reject()
        })
    })
        
    }


    encryptData(data:any, key = ''): string | undefined {

    try {
        if (key) {
            return CryptoJS.AES.encrypt(JSON.stringify(data), `${this.encryptSecretKey}?key=${key}`).toString();
        } 
    } catch (e) {
        console.log(e);
    }
    }

    decryptData(data:any, key = '') {

    try {
        if (key) {
        const bytes = CryptoJS.AES.decrypt(data, `${this.encryptSecretKey}?key=${key}`);

        if (bytes.toString()) {
            return JSON.parse(bytes.toString(CryptoJS.enc.Utf8));
        }

        return data;
        }
    } catch (e) {
        console.log(e);
    }
    }
    
}
