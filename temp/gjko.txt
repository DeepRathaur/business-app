const handleRequest = async <T>(
  method: "get" | "post",
  url: string,
  data?: any,
  isFile: boolean = false,
  isBusinessPayment: boolean = false
): Promise<T> => {
  try {
    const config = isFile ? { responseType: "blob" as const } : {};
    const configServiceId = isBusinessPayment
      ? {
          "x-service-id":
            process.env.NEXT_PUBLIC_BUSINESS_PAYMENT_BFE_SERVICE_ID,
        }
      : {};

    // const headerConfig = (isBusinessPayment)?configServiceId:config;
    const response =
      method === "post"
        ? await axiosClient.post<T>(url, data, config)
        : await axiosClient.get<T>(url + data);
    return response.data;
  } catch (error) {
    console.error(`API Error [${method.toUpperCase()} ${url}]:`, error);

    
    throw error;
  }
};

export const uploadFile = async (
  form: FormData,
  onProgress?: (percent: number) => void
): Promise<any> => {
  const url = UrlBuilder.createUrl(RENDER_URL.UPLOAD_FILE);
  try {
    const response = await axiosClient.post(url, form, {
      headers: {
        //"Content-Type": "multipart/form-data",
      },
      onUploadProgress: (progressEvent) => {
        if (progressEvent.total) {
          const percent = Math.round(
            (progressEvent.loaded * 100) / progressEvent.total
          );
          if (onProgress) onProgress(percent);
        } else {
          if (onProgress) onProgress(0); // fallback or skip
        }
      },
    });

    return response.data;
  } catch (error: any) {
    console.error("[File Upload Error]", error);
    throw error;
  }
};

import { URL } from "@/core/constant/url";

export class UrlBuilder {
    private static readonly _baseUrl: string = URL.BASE_URL;

    public static createUrl(path: string, queryParams?: Record<string, string>): string {
        let fullUrl = `${this._baseUrl}${path}`;
        if (queryParams) {
            const queryString = new URLSearchParams(queryParams).toString();
            fullUrl += `?${queryString}`;
        }
        return fullUrl;
    }
}
