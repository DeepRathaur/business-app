"use client";

import { LanguageService } from "@/core/service/LanguageService";
import { ResponseCodeEnum } from "@/shared/enum";
import React, {
  createContext,
  useState,
  useEffect,
  ReactNode,
  useMemo,
} from "react";
import en from "../../public/locales/en.json";
import fr from "../../public/locales/fr.json";

interface ConfigContextType {
  configLocale: any | null;
  loading: boolean;
  error: any;
  selectedLang: string;
  setSelectedLang: (lang: string) => void;
}

export const LocaleContext = createContext<ConfigContextType>({
  configLocale: null,
  loading: true,
  error: null,
  selectedLang: "en",
  setSelectedLang: () => {},
});

export const LocaleProvider = ({ children }: { children: ReactNode }) => {
  const [selectedLang, setSelectedLang] = useState<string>("en");
  const [configLocale, setLocale] = useState<any | null>(null);
  const [loading, setLoading] = useState<boolean>(true);
  const [error, setError] = useState<unknown>(null);
  
  const LOCALES: Record<string, any> = { en, fr };
  
  const loadLocalLocaleData = async () => {
    try {
      const localData = LOCALES[selectedLang] ?? LOCALES["en"];
      setLocale(localData);
    } catch (err) {
      console.error("Failed to set local locale:", err);
    }
  };
  
  useEffect(() => {
    const service = new LanguageService();

    const loadLocalLocale = async () => {
      try {
        const localData = await fetch(`/locales/${selectedLang}.json`).then(
          (res) => res.json()
        );
        setLocale(localData);
      } catch (err) {
        console.error("Failed to load local locale:", err);
      }
    };

    const fetchLocaleData = async () => {
      setLoading(true);
      try {
        const response = await service.fetchLocale(selectedLang);
        if (response && response.statusCode === ResponseCodeEnum.SUCCESS) {
          const data = response.result?.data;
          setLocale(data); // Override with API data
          setError(null);
        } else {
          console.warn("API did not return success, keeping local data.");
        }
      } catch (err) {
        console.error("API failed:", err);
        setError(err);
      } finally {
        setLoading(false);
      }
    };

    loadLocalLocaleData();
    fetchLocaleData(); // Then try API
  }, [selectedLang]);

  const contextValue = useMemo(
    () => ({
      configLocale,
      loading,
      error,
      selectedLang,
      setSelectedLang,
    }),
    [configLocale, loading, error, selectedLang]
  );

  return (
    <LocaleContext.Provider value={contextValue}>
      {children}
    </LocaleContext.Provider>
  );
};

import React, {
  createContext,
  useContext,
  useState,
  useEffect,
  ReactNode,
} from "react";
import { accountService } from "@/core/service/accountService";

interface AuthContextType {
  token: string | null;
  loginAuth: (token: string) => void;
  logout: () => void;
  isAuthed: boolean;
  setIsAuthed: React.Dispatch<React.SetStateAction<boolean>>;
}

const AuthContext = createContext<AuthContextType | undefined>(undefined);

interface AuthProviderProps {
  children: ReactNode;
}

export const AuthProvider: React.FC<AuthProviderProps> = ({ children }) => {
  const [token, setToken] = useState<string | null>(null);
  const [isAuthed, setIsAuthed] = useState(false);

  useEffect(() => {
    const storedToken = localStorage.getItem("token");
    if (storedToken) {
      setToken(storedToken);
      accountService.setAccessToken(storedToken);
      setIsAuthed(true);
    }
  }, []);

  useEffect(() => {
    const storedUser = accountService.getUser();
    if (storedUser) {
      setIsAuthed(true);
    } else {
      console.log('No user details found')
    }
  }, []);

  const loginAuth = (newToken: string) => {
    setToken(newToken);
    accountService.setAccessToken(newToken);
    setIsAuthed(true);
  };

  const logout = () => {
    setToken(null);
    setIsAuthed(false);
    localStorage.removeItem("token");
    window.dispatchEvent(new Event("auth-changed"));
  };

  return (
    <AuthContext.Provider value={{ token, loginAuth, logout, isAuthed, setIsAuthed }}>
      {children}
    </AuthContext.Provider>
  );
};

export function useAuth(): AuthContextType {
  const context = useContext(AuthContext);
  if (!context) {
    throw new Error("useAuth must be used within an AuthProvider");
  }
  return context;
}



"use client";

import { AccountDetailsModel } from "@/shared/models";
import React, {
  createContext, useReducer, useMemo, useEffect, useCallback, useContext,
} from "react";

export type Account = { name: string; number: string };

// ---- SINGLETON on globalThis ----
type CtxShape =
  | {
      ecareaccounts: AccountDetailsModel[];
      selectedIndex: number;
      selected: AccountDetailsModel | null;
      select: (i: number) => void;
      loadMore?: () => void;
      isLoadingMore?: boolean;
      hasMore?: boolean;
      totalRecords?:number;
    }
  | null;

const CTX_KEY = "__ACCOUNT_CONTEXT_SINGLETON__";
const Existing =
  (globalThis as any)[CTX_KEY] as React.Context<CtxShape> | undefined;

const AccountContext = Existing ?? createContext<CtxShape>(null);
if (!(globalThis as any)[CTX_KEY]) (globalThis as any)[CTX_KEY] = AccountContext;

// ---- Reducer + Provider ----
const STORAGE_KEY = "defaultAccountNumber";

type State = { ecareaccounts: AccountDetailsModel[]; selectedIndex: number };
type Action =
  | { type: "select"; index: number }
  | { type: "hydrateByNumber"; number: string }
  | { type: "setAccounts"; ecareaccounts: AccountDetailsModel[] };

function reducer(state: State, action: Action): State {
  switch (action.type) {
    case "select":
      return { ...state, selectedIndex: action.index };
    case "hydrateByNumber": {
      const idx = state.ecareaccounts.findIndex((a) => a.accountNo === action.number);
      return idx === -1 ? state : { ...state, selectedIndex: idx };
    }
    case "setAccounts": {
      const currentNumber = state.ecareaccounts[state.selectedIndex]?.accountNo;
      const newIndex = action.ecareaccounts.findIndex((a) => a.accountNo === currentNumber);
      return { ecareaccounts: action.ecareaccounts, selectedIndex: newIndex === -1 ? 0 : newIndex };
    }
    default:
      return state;
  }
}

export function AccountProvider({
  ecareaccounts,
  initialIndex = 0,
  loadMore,
  isLoadingMore,
  hasMore,
  children,
  totalRecords,
}: {
  ecareaccounts: AccountDetailsModel[];
  initialIndex?: number;
  loadMore?: () => void;
  isLoadingMore:boolean,
  hasMore:boolean,
  children: React.ReactNode;
  totalRecords:number;
}) {
  // console.log("acounts=====",accounts)
  const [state, dispatch] = useReducer(reducer, { ecareaccounts, selectedIndex: initialIndex });

  useEffect(() => { dispatch({ type: "setAccounts", ecareaccounts }); }, [ecareaccounts]);

  useEffect(() => {
    try {
      const saved = typeof window !== "undefined" && localStorage.getItem(STORAGE_KEY);
      if (saved) dispatch({ type: "hydrateByNumber", number: JSON.parse(saved as string) });
    } catch {}
  }, [ecareaccounts]);

  const select = useCallback((i: number) => {
    dispatch({ type: "select", index: i });
    const acc = ecareaccounts[i];
    if (acc) {
      try { localStorage.setItem(STORAGE_KEY, JSON.stringify(acc.accountNo)); } catch {}
      if (typeof window !== "undefined") {
        window.dispatchEvent(new CustomEvent("default-account-changed", { detail: acc }));
      }
    }
  }, [ecareaccounts]);

  const selected = state.ecareaccounts[state.selectedIndex] ?? null;
  const value = useMemo(() => ({ ...state, selected, select, loadMore, isLoadingMore, hasMore, totalRecords}), [state, selected, select, loadMore, isLoadingMore, hasMore, totalRecords]);

  return <AccountContext.Provider value={value}>{children}</AccountContext.Provider>;
}

export function useAccount() {
  const ctx = useContext(AccountContext);
  if (!ctx) {
    throw new Error(
      "useAccount must be used within AccountProvider (make sure app/layout.tsx wraps with Providers and all imports use)."
    );
  }
  return ctx;
}

export function useSelectedAccount() {
  const { selected } = useAccount();
  return selected;
}

import React, { createContext, useContext, useState, useCallback, useEffect } from "react";

type Toast = {
  id: number;
  message: string;
  type?: "success" | "error" | "info";
};

type ToastContextType = {
  showToast: (message: string, type?: Toast["type"]) => void;
};

const ToastContext = createContext<ToastContextType | undefined>(undefined);

let toastId = 0;

export const ToastProvider: React.FC<{ children: React.ReactNode }> = ({ children }) => {
  const [toasts, setToasts] = useState<Toast[]>([]);

  const showToast = useCallback((message: string, type: Toast["type"] = "info") => {
    const id = toastId++;
    setToasts((prev) => [...prev, { id, message, type }]);

    // Auto-remove after 3 seconds
    setTimeout(() => {
      setToasts((prev) => prev.filter((t) => t.id !== id));
    }, 3000);
  }, []);

  return (
    <ToastContext.Provider value={{ showToast }}>
      {children}

      <div className="fixed bottom-5 left-1/2 transform -translate-x-1/2 flex flex-col space-y-2 z-50 max-w-xs w-full px-2">
        {toasts.map(({ id, message, type }) => (
          <div
            key={id}
            className={`rounded-md px-4 py-3 shadow-md text-white
              ${
                type === "success"
                  ? "bg-green-500"
                  : type === "error"
                  ? "bg-red-500"
                  : "bg-blue-500"
              }
              `}
          >
            {message}
          </div>
        ))}
      </div>
    </ToastContext.Provider>
  );
};

export const useToast = (): ToastContextType => {
  const context = useContext(ToastContext);
  if (!context) {
    throw new Error("useToast must be used within a ToastProvider");
  }
  return context;
};

"use client";

import { usePathname } from "next/navigation";
import { routeSettings } from "@/config/routeConfig";

export function getRouteSettings(pathname: string) {
  if (pathname.includes('/checkout') || pathname.includes('/payment-intigration')) {
    return {
      hideHeader: true,
      showFooter: false,
      forceHideHeader: true,
    };
  }

  if (pathname.startsWith("/manage-service/form-renderer")) {
    return {
      hideHeader: true,
      showFooter: false,
      forceHideHeader: true
    };
  }

  return routeSettings[pathname] ?? {
    hideHeader: false,
    showFooter: false,
  };
}

export default function RouteSettingsProvider({
  children
}: {
  children: (settings: ReturnType<typeof getRouteSettings>, pathname: any) => React.ReactNode;
}) {
  const pathname = usePathname();
  const settings = getRouteSettings(pathname ?? "");

  return <>{children(settings, pathname)}</>;
}
